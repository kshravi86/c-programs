Solution : "Between Two Sets"

#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

int getTotalX(int a_size, int* a, int b_size, int* b, int min, int max) {
    int flag = 0, count = 0;
    for(int i = min; i <= max; i++) {
        flag = 0;
        for(int j = 0; j < a_size; j++) {
            if(i % a[j] != 0) {
                flag = 1;
                break;
            }
        }
        if(flag == 1)
            continue;
        else {
            for(int j = 0; j < b_size; j++) {
                if(b[j] % i != 0) {
                    flag = 1;
                    break;
                }
            }
            if(flag == 0)
                count++;
        }
    }
    return count;
}

int main() {
    int n; 
    int m; 
    scanf("%d %d", &n, &m);
    int *a = malloc(sizeof(int) * n);
    int max = 0;
    int min = 0;
    for(int a_i = 0; a_i < n; a_i++){
        scanf("%d",&a[a_i]);
        if(a_i == 0) {
            max = a[a_i];
            min = a[a_i];
        }
        else {
            if(a[a_i] > max)
                max = a[a_i];
            if(a[a_i] < min)
                min = a[a_i];
        }
    }
    int *b = malloc(sizeof(int) * m);
    for(int b_i = 0; b_i < m; b_i++){
        scanf("%d",&b[b_i]);
        if(b[b_i] > max)
            max = b[b_i];
        if(b[b_i] < min)
            min = b[b_i];
    }
    int total = getTotalX(n, a, m, b, min, max);
    printf("%d\n", total);
    return 0;
}
