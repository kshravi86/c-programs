/*
---------------------------------
Solution : "Breaking The Records"
---------------------------------
*/

#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

int main() {
    int n; 
    scanf("%d",&n);
    int *s = malloc(sizeof(int) * n);
    int max, min, maxbreak = 0, minbreak = 0;
    for(int s_i = 0; s_i < n; s_i++){
        scanf("%d",&s[s_i]);
        if(s_i == 0) {
            max = s[s_i];
            min = s[s_i];
        }
        else {
            if(s[s_i] > max) {
                maxbreak++;
                max = s[s_i];
            }
            if(s[s_i] < min) {
                minbreak++;
                min = s[s_i];
            }
        }
    }
    printf("%d %d", maxbreak, minbreak);
    return 0;
}
