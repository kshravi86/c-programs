#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

int solve(int n, int* s, int d, int m) {
    int pos = 0, count = 0, sum;
    while(pos + m <= n) {
        sum = 0;
        for(int i = pos; i < (pos + m) ; i++)
            sum += s[i];
        if(sum == d)
            count++;
        pos++;
    }
    return count;
}

int main() {
    int n; 
    scanf("%d", &n);
    int *s = malloc(sizeof(int) * n);
    for(int s_i = 0; s_i < n; s_i++){
       scanf("%d",&s[s_i]);
    }
    int d, m; 
    scanf("%d %d", &d, &m);
    int result = solve(n, s, d, m);
    printf("%d\n", result);
    return 0;
}
